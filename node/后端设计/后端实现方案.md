## 1. 技术栈与架构

### 1.1 技术选型

**核心框架**
- **运行时**: Node.js >= 16.0
- **Web 框架**: Express 4.x
- **数据库**: MySQL 8.0+
- **缓存/队列**: Redis 6.0+（可选，强烈推荐）
- **ORM**: Sequelize 或 mysql2（原生 SQL）

**工具库**
- **认证**: jsonwebtoken (JWT)
- **密码加密**: bcrypt
- **参数校验**: express-validator 或 joi
- **文件上传**: multer
- **日志**: winston 或 pino
- **环境变量**: dotenv
- **任务队列**: Bull（基于 Redis，可选）

### 1.2 架构设计

```
┌─────────────┐
│  前端(Vue)  │
└──────┬──────┘
       │ HTTP/HTTPS
┌──────▼─────────────────────────────────────┐
│         Express API 服务层                 │
│  ┌──────────┐  ┌──────────┐  ┌──────────┐ │
│  │ 路由层   │  │ 控制器层  │  │ 服务层   │ │
│  └──────────┘  └──────────┘  └──────────┘ │
│  ┌──────────┐  ┌──────────┐  ┌──────────┐ │
│  │ 中间件   │  │ 数据访问层│  │ 工具层   │ │
│  └──────────┘  └──────────┘  └──────────┘ │
└──────┬─────────────────────────────────────┘
       │
┌──────▼──────┐  ┌──────────┐
│   MySQL     │  │  Redis   │
│   数据库     │  │   缓存    │
└─────────────┘  └──────────┘
```

**分层说明**
- **路由层 (Routes)**: 定义 API 路由，绑定中间件
- **控制器层 (Controllers)**: 处理请求/响应，参数校验
- **服务层 (Services)**: 业务逻辑，事务管理
- **数据访问层 (Models/Repositories)**: 数据库操作
- **中间件 (Middleware)**: 认证、限流、错误处理等

### 1.3 模块划分

1. **auth** - 认证模块（注册、登录、锁定）
2. **users** - 用户模块（资料、统计）
3. **catalog** - 内容模块（歌曲、歌手、专辑、类型）
4. **playlists** - 歌单模块
5. **playback** - 播放模块（行为采集、历史）
6. **ratings** - 评分模块
7. **comments** - 评论模块
8. **notifications** - 通知模块
9. **recommend** - 推荐模块
10. **admin** - 管理后台模块

### 2.2 关键文件示例

#### app.js（应用入口）

```javascript
const express = require('express');
const cors = require('cors');
const helmet = require('helmet');
const morgan = require('morgan');
const { errorHandler } = require('./middleware/errorHandler');
const routes = require('./routes');

const app = express();

// 安全中间件
app.use(helmet());
app.use(cors({
  origin: process.env.CORS_ORIGIN || 'http://localhost:3000',
  credentials: true
}));

// 日志
app.use(morgan('combined'));

// 解析请求体
app.use(express.json({ limit: '10mb' }));
app.use(express.urlencoded({ extended: true, limit: '10mb' }));

// 路由
app.use('/api', routes);

// 健康检查
app.get('/health', (req, res) => {
  res.json({ status: 'ok', timestamp: new Date().toISOString() });
});

// 错误处理（必须放在最后）
app.use(errorHandler);

module.exports = app;
```

#### utils/response.js（统一响应格式）

```javascript
/**
 * 统一响应格式
 */
class ResponseHelper {
  static success(res, data = null, message = 'success') {
    return res.status(200).json({
      code: 0,
      message,
      data
    });
  }

  static error(res, code, message, statusCode = 400) {
    return res.status(statusCode).json({
      code,
      message,
      data: null
    });
  }

  static pagination(res, list, total, page, pageSize) {
    return res.status(200).json({
      code: 0,
      message: 'success',
      data: {
        list,
        total,
        page: parseInt(page),
        pageSize: parseInt(pageSize),
        totalPages: Math.ceil(total / pageSize)
      }
    });
  }
}

module.exports = ResponseHelper;
```

## 3. 数据库设计

### 3.1 核心表结构（完整 SQL）

#### 3.1.1 用户表（users）

```sql
CREATE TABLE users (
    id INT PRIMARY KEY AUTO_INCREMENT,
    username VARCHAR(50) UNIQUE NOT NULL COMMENT '用户名',
    password_hash VARCHAR(255) NOT NULL COMMENT '密码哈希',
    email VARCHAR(100) UNIQUE NOT NULL COMMENT '邮箱',
    nickname VARCHAR(50) COMMENT '昵称',
    avatar VARCHAR(500) COMMENT '头像URL',
    gender TINYINT DEFAULT 0 COMMENT '性别: 0-未知, 1-男, 2-女',
    birthday DATE COMMENT '生日',
    status TINYINT DEFAULT 1 COMMENT '状态: 0-封禁, 1-正常',
    last_login_at DATETIME COMMENT '最后登录时间',
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    INDEX idx_username (username),
    INDEX idx_email (email),
    INDEX idx_status (status)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='用户表';
```

### 5.1 登录锁定机制

#### services/authService.js

```javascript
const bcrypt = require('bcrypt');
const jwt = require('jsonwebtoken');
const db = require('../config/database');
const { ERROR_CODES } = require('../utils/constants');

class AuthService {
  /**
   * 用户登录（含锁定机制）
   */
  async login(identity, password, ip) {
    const connection = await db.getConnection();
    
    try {
      await connection.beginTransaction();
      
      // 1. 检查锁定状态
      const [attempts] = await connection.query(
        `SELECT * FROM user_login_attempts 
         WHERE identity = ? AND (locked_until IS NULL OR locked_until > NOW())`,
        [identity]
      );
      
      if (attempts.length > 0 && attempts[0].locked_until) {
        await connection.rollback();
        return {
          code: ERROR_CODES.AUTH_LOCKED,
          message: '账号已锁定，请30分钟后重试',
          data: { lockedUntil: attempts[0].locked_until }
        };
      }
      
      // 2. 查找用户（支持用户名或邮箱）
      const [users] = await connection.query(
        `SELECT * FROM users 
         WHERE (username = ? OR email = ?) AND status = 1`,
        [identity, identity]
      );
      
      if (users.length === 0) {
        await this.recordLoginFailure(connection, identity, ip);
        await connection.commit();
        return {
          code: ERROR_CODES.AUTH_INVALID_CREDENTIALS,
          message: '账号或密码错误'
        };
      }
      
      const user = users[0];
      
      // 3. 验证密码
      const isValid = await bcrypt.compare(password, user.password_hash);
      
      if (!isValid) {
        await this.recordLoginFailure(connection, identity, ip);
        await connection.commit();
        return {
          code: ERROR_CODES.AUTH_INVALID_CREDENTIALS,
          message: '账号或密码错误'
        };
      }
      
      // 4. 登录成功，清除失败记录
      await connection.query(
        `DELETE FROM user_login_attempts WHERE identity = ?`,
        [identity]
      );
      
      // 5. 更新最后登录时间
      await connection.query(
        `UPDATE users SET last_login_at = NOW() WHERE id = ?`,
        [user.id]
      );
      
      // 6. 生成 JWT Token
      const token = jwt.sign(
        { userId: user.id, username: user.username },
        process.env.JWT_SECRET,
        { expiresIn: '7d' }
      );
      
      await connection.commit();
      
      return {
        code: ERROR_CODES.SUCCESS,
        message: '登录成功',
        data: {
          token,
          userInfo: {
            id: user.id,
            username: user.username,
            nickname: user.nickname,
            avatar: user.avatar,
            email: user.email
          }
        }
      };
      
    } catch (error) {
      await connection.rollback();
      throw error;
    } finally {
      connection.release();
    }
  }
  
  /**
   * 记录登录失败
   */
  async recordLoginFailure(connection, identity, ip) {
    const [attempts] = await connection.query(
      `SELECT * FROM user_login_attempts WHERE identity = ?`,
      [identity]
    );
    
    if (attempts.length === 0) {
      // 首次失败
      await connection.query(
        `INSERT INTO user_login_attempts (identity, ip, fail_count, last_failed_at) 
         VALUES (?, ?, 1, NOW())`,
        [identity, ip]
      );
    } else {
      const attempt = attempts[0];
      const newFailCount = attempt.fail_count + 1;
      let lockedUntil = null;
      
      // 达到3次失败，锁定30分钟
      if (newFailCount >= 3) {
        lockedUntil = new Date(Date.now() + 30 * 60 * 1000);
      }
      
      await connection.query(
        `UPDATE user_login_attempts 
         SET fail_count = ?, locked_until = ?, last_failed_at = NOW() 
         WHERE identity = ?`,
        [newFailCount, lockedUntil, identity]
      );
    }
  }
}

module.exports = new AuthService();
```

```javascript
/**
 * 离线推荐计算任务（每天凌晨执行）
 */
async function calculateRecommendations() {
  console.log('开始计算推荐...');
  
  // 1. 获取所有活跃用户
  const [users] = await db.query(
    `SELECT id FROM users WHERE status = 1 
     AND last_login_at >= DATE_SUB(NOW(), INTERVAL 30 DAY)`
  );
  
  for (const user of users) {
    try {
      // 2. 基于用户的协同过滤
      const userBasedResults = await userBasedRecommend(user.id, 20, 50);
      
      // 3. 基于物品的协同过滤
      const itemBasedResults = await itemBasedRecommend(user.id, 50);
      
      // 4. 混合推荐
      const recommendations = mergeRecommendations(
        userBasedResults,
        itemBasedResults,
        50
      );
      
      // 5. 保存推荐结果
      await saveRecommendations(user.id, 'song', recommendations);
      
      console.log(`用户 ${user.id} 推荐计算完成`);
    } catch (error) {
      console.error(`用户 ${user.id} 推荐计算失败:`, error);
    }
  }
  
  console.log('推荐计算完成');
}

/**
 * 合并推荐结果
 */
function mergeRecommendations(userBased, itemBased, limit) {
  const scoreMap = {};
  
  // 合并基于用户的推荐（权重0.6）
  userBased.forEach(item => {
    if (!scoreMap[item.songId]) {
      scoreMap[item.songId] = 0;
    }
    scoreMap[item.songId] += item.predictedRating * 0.6;
  });
  
  // 合并基于物品的推荐（权重0.4）
  itemBased.forEach(item => {
    if (!scoreMap[item.songId]) {
      scoreMap[item.songId] = 0;
    }
    scoreMap[item.songId] += item.predictedRating * 0.4;
  });
  
  // 排序并返回TopN
  return Object.entries(scoreMap)
    .map(([songId, score]) => ({
      songId: Number(songId),
      score
    }))
    .sort((a, b) => b.score - a.score)
    .slice(0, limit);
}

/**
 * 保存推荐结果到数据库
 */
async function saveRecommendations(userId, itemType, recommendations) {
  const connection = await db.getConnection();
  
  try {
    await connection.beginTransaction();
    
    // 删除旧推荐
    await connection.query(
      `DELETE FROM recommendation_items 
       WHERE user_id = ? AND item_type = ?`,
      [userId, itemType]
    );
    
    // 插入新推荐
    const values = recommendations.map((rec, index) => [
      userId,
      itemType,
      rec.songId,
      rec.score,
      generateReason(rec, index),
      'v1'
    ]);
    
    await connection.query(
      `INSERT INTO recommendation_items 
       (user_id, item_type, item_id, score, reason, algo_version) 
       VALUES ?`,
      [values]
    );
    
    await connection.commit();
  } catch (error) {
    await connection.rollback();
    throw error;
  } finally {
    connection.release();
  }
}

function generateReason(rec, index) {
  if (index < 10) {

    return '喜欢这首歌的用户也喜欢';
  } else {
    return '来自您常听的歌手/类型';
  }
}
```

```javascript
class RecommendService {

   */
  async getPersonalRecommendations(userId, type = 'song', limit = 20) {

    if (recommendations.length < limit) {
      const coldStart = await this.getColdStartRecommendations(
        userId,
        type,
        limit - recommendations.length
      );
      recommendations.push(...coldStart);
    }
    
    // 3. 过滤已听/已收藏的歌曲

    return {
      code: 0,
      message: 'success',

              COALESCE(SUM(ssd.play_count), 0) as total_plays
       FROM songs s
       INNER JOIN artists a ON s.artist_id = a.id
       LEFT JOIN song_stats_daily ssd ON s.id = ssd.song_id 
         AND ssd.stat_date >= DATE_SUB(CURDATE(), INTERVAL 7 DAY)
       WHERE s.status = 1
       GROUP BY s.id
       ORDER BY total_plays DESC, s.created_at DESC
       LIMIT ?`,
      [limit]
    );
    
    return hotSongs.map(song => ({
      songId: song.id,
      song: {
        id: song.id,
        title: song.title,
        coverUrl: song.cover_url,
        artist: { id: song.artist_id, name: song.artist_name }
      },
      score: 0.5, // 默认分数
      reason: '热门推荐'
    }));
  }
  
  /**
   * 过滤已听/已收藏的歌曲
   */
  async filterListenedSongs(userId, recommendations) {
    const songIds = recommendations.map(r => r.item_id || r.songId);
    
    if (songIds.length === 0) return recommendations;
    
    // 查询已听/已收藏
    const [listened] = await db.query(
      `SELECT DISTINCT song_id FROM play_history WHERE user_id = ?
       UNION
       SELECT DISTINCT song_id FROM user_favorites WHERE user_id = ?`,
      [userId, userId]
    );
    
    const listenedSet = new Set(listened.map(r => r.song_id));
    
    return recommendations.filter(r => {
      const songId = r.item_id || r.songId;
      return !listenedSet.has(songId);
    });
  }

       GROUP BY s2.id
       ORDER BY COUNT(*) DESC
       LIMIT ?`,
      [songId, songId, limit]
    );
    
    return similar;
  }
```
